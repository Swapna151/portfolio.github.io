Business Case Overview

Smart Energy Management directly supports **SDG 7** by providing reliable, clean, and affordable energy through smart grids and renewable integration. It also advances **SDG 11, 12, and 13** by enabling efficient urban energy use, responsible consumption, and reducing carbon emissions.

Step 1: Understanding City Energy Demand

Analyze zone-wise consumption patterns, identify peak load periods, and map high-pressure areas for informed planning and resource allocation.

Step 2: Local Renewable Energy Generation

Deploy rooftop solar panels, small wind units, and community renewable systems to generate cleaner energy locally and reduce dependency on main grids.

Step 3: Citywide Battery Storage Systems

Install large-scale storage units to capture excess energy produced during the day and release it during peak-demand hours or grid instability.

Step 4: District-Level Microgrids

Build independent microgrids for key districts to improve resilience, minimize outages, and reduce transmission losses.

Step 5: Smart Monitoring & Smart Meters

Track real-time usage, detect faults early, prevent overloads, and enable user-level consumption optimization.

Step 6: AI-Based Load Balancing

Use AI models to forecast demand, automatically regulate supply, and stabilize the grid across all city zones.

    #include < iostream >
using namespace std;

struct Node {
    int data, height;
    Node *left, *right;
    Node(int x) : data(x), height(1), left(NULL), right(NULL) {}
};

int h(Node* n) { return n ? n->height : 0; }
int bal(Node* n) { return n ? h(n->left) - h(n->right) : 0; }

Node* rRotate(Node* y) {
    Node* x = y->left;
    y->left = x->right;
    x->right = y;
    y->height = max(h(y->left), h(y->right)) + 1;
    x->height = max(h(x->left), h(x->right)) + 1;
    return x;
}

Node* lRotate(Node* x) {
    Node* y = x->right;
    x->right = y->left;
    y->left = x;
    x->height = max(h(x->left), h(x->right)) + 1;
    y->height = max(h(y->left), h(y->right)) + 1;
    return y;
}

Node* insert(Node* r, int k) {
    if (!r) return new Node(k);
    if (k < r->data) r->left = insert(r->left, k);
    else if (k > r->data) r->right = insert(r->right, k);
    else return r;

    r->height = 1 + max(h(r->left), h(r->right));
    int b = bal(r);

    if (b > 1 && k < r->left->data) return rRotate(r);
    if (b < -1 && k > r->right->data) return lRotate(r);
    if (b > 1 && k > r->left->data) { r->left = lRotate(r->left); return rRotate(r); }
    if (b < -1 && k < r->right->data) { r->right = rRotate(r->right); return lRotate(r); }

    return r;
}

void heapBottomUp(int H[], int n) {
    for (int k = n / 2; k >= 1; k--) {
        int v = H[k];
        int heap = 0;

        while (!heap && 2 * k <= n) {
            int j = 2 * k;

            if (j < n && H[j] < H[j + 1])
                j = j + 1;

            if (v >= H[j])
                heap = 1;
            else {
                H[k] = H[j];
                k = j;
            }
        }
        H[k] = v;
    }
}

#include < bits/stdc++.h >
using namespace std;

class SegmentTree {
    int n;
    vector st;

public:
    SegmentTree(vector& a) {
        n = a.size();
        st.resize(4 * n);
        build(1, 0, n - 1, a);
    }

    void build(int node, int l, int r, vector& a) {
        if (l == r) {
            st[node] = a[l];
            return;
        }
        int m = (l + r) / 2;
        build(2*node, l, m, a);
        build(2*node+1, m+1, r, a);
        st[node] = st[2*node] + st[2*node+1];
    }

    int query(int node, int l, int r, int ql, int qr) {
        if (qr < l || ql > r) return 0;
        if (ql <= l && r <= qr) return st[node];
        int m = (l + r) / 2;
        return query(2*node, l, m, ql, qr) +
               query(2*node+1, m+1, r, ql, qr);
    }

    void update(int node, int l, int r, int idx, int val) {
        if (l == r) {
            st[node] = val;
            return;
        }
        int m = (l + r) / 2;
        if (idx <= m) update(2*node, l, m, idx, val);
        else update(2*node+1, m+1, r, idx, val);
        st[node] = st[2*node] + st[2*node+1];
    }

    int rangeQuery(int l, int r) {
        return query(1, 0, n - 1, l, r);
    }

    void pointUpdate(int idx, int val) {
        update(1, 0, n - 1, idx, val);
    }
};

#include < iostream >
using namespace std;

class UnionFind {
    int parent[1000], rank[1000];

public:
    UnionFind(int n) {
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            rank[i] = 0;
        }
    }

    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]); // path compression
        return parent[x];
    }

    void unite(int x, int y) {
        int px = find(x);
        int py = find(y);

        if (px == py) return;

        if (rank[px] < rank[py])
            parent[px] = py;
        else if (rank[px] > rank[py])
            parent[py] = px;
        else {
            parent[py] = px;
            rank[px]++;
        }
    }
};

int main() {
    UnionFind uf(5);

    uf.unite(0, 1);
    uf.unite(1, 2);
    uf.unite(3, 4);

    if (uf.find(0) == uf.find(2))
        cout << "0 and 2 are connected\n";
    else
        cout << "0 and 2 are not connected\n";

    return 0;
}

void heapBottomUp(int H[], int n) {
    for (int k = n / 2; k >= 1; k--) {
        int v = H[k];

#include < iostream>
using namespace std;

struct Edge {
    int u, v, w;
};

void bubbleSort(Edge edges[], int m) {
    for (int i = 0; i < m - 1; i++) {
        for (int j = 0; j < m - i - 1; j++) {
            if (edges[j].w > edges[j + 1].w) {
                Edge temp = edges[j];
                edges[j] = edges[j + 1];
                edges[j + 1] = temp;
            }
        }
    }
}

// Simple union (not optimized ‚Äî acceptable for DAA basics)
void Union(int parent[], int n, int u, int v) {
    int temp = parent[u];
    for (int i = 0; i < n; i++) {
        if (parent[i] == temp)
            parent[i] = parent[v];
    }
}

bool Find(int parent[], int u, int v) {
    return parent[u] == parent[v];
}

void kruskals(int n, int m, Edge edges[]) {
    int parent[100];
    for (int i = 0; i < n; i++)
        parent[i] = i;

    bubbleSort(edges, m);

    int mstWeight = 0;
    cout << "\nEdges in MST:\n";

    for (int i = 0; i < m; i++) {
        int u = edges[i].u;
        int v = edges[i].v;
        int w = edges[i].w;

        if (!Find(parent, u, v)) {
            cout << u << " -- " << v << " == " << w << endl;
            mstWeight += w;
            Union(parent, n, u, v);
        }
    }

    cout << "Total Weight of MST = " << mstWeight << endl;
}

int main() {
    int n, m;
    cout << "Enter number of vertices and edges: ";
    cin >> n >> m;

    Edge edges[100];
    cout << "Enter edges as: u v w\n";
    for (int i = 0; i < m; i++)
        cin >> edges[i].u >> edges[i].v >> edges[i].w;

    kruskals(n, m, edges);
    return 0;
}

// hash Table
#include 
using namespace std;

#define SIZE 10

class Node {
public:
    int key;
    Node* next;
    Node(int k) : key(k), next(NULL) {}
};

class HashTable {
    Node* table[SIZE];

public:
    HashTable() {
        for (int i = 0; i < SIZE; i++)
            table[i] = NULL;
    }

    int hashFunction(int key) {
        return key % SIZE;
    }

    void insert(int key) {
        int idx = hashFunction(key);
        Node* newNode = new Node(key);
        newNode->next = table[idx];
        table[idx] = newNode;
    }

    bool search(int key) {
        int idx = hashFunction(key);
        Node* temp = table[idx];
        while (temp) {
            if (temp->key == key)
                return true;
            temp = temp->next;
        }
        return false;
    }

    void display() {
        for (int i = 0; i < SIZE; i++) {
            cout << i << ": ";
            Node* temp = table[i];
            while (temp) {
                cout << temp->key << " -> ";
                temp = temp->next;
            }
            cout << "NULL\n";
        }
    }
};

int main() {
    HashTable ht;

    ht.insert(10);
    ht.insert(20);
    ht.insert(15);
    ht.insert(25);
    ht.insert(30);

    ht.display();

    cout << "\nSearch 15: " << (ht.search(15) ? "Found" : "Not Found");
    return 0;
}

    Node(int k) : key(k), height(1), left(NULL), right(NULL) {}
};

int h(Node* n) {
    return n ? n->height : 0;
}

int balance(Node* n) {
    return n ? h(n->left) - h(n->right) : 0;
}

Node* rightRotate(Node* y) {
    Node* x = y->left;
    Node* T2 = x->right;

    x->right = y;
    y->left = T2;

    y->height = max(h(y->left), h(y->right)) + 1;
    x->height = max(h(x->left), h(x->right)) + 1;

    return x;
}

Node* leftRotate(Node* x) {
    Node* y = x->right;
    Node* T2 = y->left;

    y->left = x;
    x->right = T2;

    x->height = max(h(x->left), h(x->right)) + 1;
    y->height = max(h(y->left), h(y->right)) + 1;

    return y;
}

Node* insert(Node* node, int key) {
    if (!node) return new Node(key);

    if (key < node->key)
        node->left = insert(node->left, key);
    else if (key > node->key)
        node->right = insert(node->right, key);
    else
        return node; // no duplicates

    node->height = 1 + max(h(node->left), h(node->right));
    int b = balance(node);

    // LL
    if (b > 1 && key < node->left->key)
        return rightRotate(node);

    // RR
    if (b < -1 && key > node->right->key)
        return leftRotate(node);

    // LR
    if (b > 1 && key > node->left->key) {
        node->left = leftRotate(node->left);
        return rightRotate(node);
    }

    // RL
    if (b < -1 && key < node->right->key) {
        node->right = rightRotate(node->right);
        return leftRotate(node);
    }

    return node;
}

void inorder(Node* root) {
    if (root) {
        inorder(root->left);
        cout << root->key << " ";
        inorder(root->right);
    }
}

        root = new TrieNode();
    }

        TrieNode* cur = root;
        for (char c : word) {
            int idx = c - 'a';
            if (!cur->child[idx])
                cur->child[idx] = new TrieNode();
            cur = cur->child[idx];
        }
        cur->isEnd = true;
    }

    bool search(string word) {
        TrieNode* cur = root;
        for (char c : word) {
            int idx = c - 'a';
            if (!cur->child[idx])
                return false;
            cur = cur->child[idx];
        }
        return cur->isEnd;
    }

    bool startsWith(string prefix) {
        TrieNode* cur = root;
        for (char c : prefix) {
            int idx = c - 'a';
            if (!cur->child[idx])
                return false;
            cur = cur->child[idx];
        }
        return true;
    }
};

        if (size == 0) return -1;

        int min = H[1];
        int last = H[size--];

            H[i] = H[j];
            i = j;
            j = 2 * i;
        }
        H[i] = last;
        return min;
    }

        for (int i = 1; i <= size; i++)
            cout << H[i] << " ";
        cout << endl;
    }

// Merge Sort
#include 
using namespace std;

// -------------------------------------------------------
// Merge() : merges two sorted arrays B and C into A
// -------------------------------------------------------
void Merge(int B[], int p, int C[], int q, int A[]) {
    int i = 0, j = 0, k = 0;

    while (i < p && j < q) {
        if (B[i] <= C[j]) {
            A[k] = B[i];
            i = i + 1;
        } else {
            A[k] = C[j];
            j = j + 1;
        }
        k = k + 1;
    }

    // If elements remain in C
    if (i == p) {
        while (j < q) {
            A[k] = C[j];
            j++;
            k++;
        }
    }
    // If elements remain in B
    else {
        while (i < p) {
            A[k] = B[i];
            i++;
            k++;
        }
    }
}

// -------------------------------------------------------
// MergeSort() : recursive merge sort following your notes
// -------------------------------------------------------
void MergeSort(int A[], int n) {
    if (n > 1) {
        int mid = n / 2;

        // Create B and C arrays
        int p = mid;
        int q = n - mid;

        int B[p];
        int C[q];

        // Copy A[0 ... mid-1] to B
        for (int i = 0; i < p; i++)
            B[i] = A[i];

        // Copy A[mid ... n-1] to C
        for (int i = 0; i < q; i++)
            C[i] = A[mid + i];

        // Recursive calls
        MergeSort(B, p);
        MergeSort(C, q);

        // Merge sorted B and C into A
        Merge(B, p, C, q, A);
    }
}

    int A[] = {9, 4, 7, 2, 6, 3, 1};

    MergeSort(A, n);

// Quick Sort
#include 
using namespace std;

// ------------------------------------------------------------
// PARTITION(A[l..r])  -- Hoare partition as per textbook
// ------------------------------------------------------------
int Partition(int A[], int l, int r) {
    int p = A[l];      // pivot
    int i = l;         // i starts at l
    int j = r + 1;     // j starts at r+1

    while (true) {

        // repeat i = i + 1 until A[i] >= p
        do {
            i++;
        } while (i <= r && A[i] < p);

        // repeat j = j - 1 until A[j] <= p
        do {
            j--;
        } while (A[j] > p);

        // if i >= j, stop
        if (i >= j)
            break;

        // swap A[i] and A[j]
        swap(A[i], A[j]);
    }

    // swap pivot A[l] with A[j]
    swap(A[l], A[j]);

    return j;   // return pivot index
}

// ------------------------------------------------------------
// QUICKSORT(A[l..r])
// ------------------------------------------------------------
void QuickSort(int A[], int l, int r) {
    if (l < r) {
        int s = Partition(A, l, r);
        QuickSort(A, l, s - 1);
        QuickSort(A, s + 1, r);
    }
}

// ------------------------------------------------------------
int main() {
    int A[] = {0, 8, 3, 1, 7, 0, 10, 2}; 
    int n = 7;   // ignoring index 0 to match algorithm style

    QuickSort(A, 1, n);

    cout << "Sorted Array: ";
    for (int i = 1; i <= n; i++)
        cout << A[i] << " ";

    return 0;
}

// Heap Sort

void heapify(int a[], int n, int i) {
    int largest = i;
    int l = 2*i + 1;
    int r = 2*i + 2;

    if (l < n && a[l] > a[largest])
        largest = l;
    if (r < n && a[r] > a[largest])
        largest = r;

    if (largest != i) {
        swap(a[i], a[largest]);
        heapify(a, n, largest);
    }
}

void heapSort(int a[], int n) {
    // Step 1: build max heap
    for (int i = n/2 - 1; i >= 0; i--)
        heapify(a, n, i);

    // Step 2: delete max one by one
    for (int i = n - 1; i > 0; i--) {
        swap(a[0], a[i]);
        heapify(a, i, 0);
    }
}

// Segment Tree (Range Sum)
#include 
using namespace std;

struct Node {
    int data, height;
    Node *left, *right;
    Node(int x) : data(x), height(1), left(NULL), right(NULL) {}
};

int h(Node* n) { return n ? n->height : 0; }
int bal(Node* n) { return n ? h(n->left) - h(n->right) : 0; }

Node* rRotate(Node* y) {
    Node* x = y->left;
    y->left = x->right;
    x->right = y;
    y->height = max(h(y->left), h(y->right)) + 1;
    x->height = max(h(x->left), h(x->right)) + 1;
    return x;
}

Node* lRotate(Node* x) {
    Node* y = x->right;
    x->right = y->left;
    y->left = x;
    x->height = max(h(x->left), h(x->right)) + 1;
    y->height = max(h(y->left), h(y->right)) + 1;
    return y;
}

Node* insert(Node* r, int k) {
    if (!r) return new Node(k);
    if (k < r->data) r->left = insert(r->left, k);
    else if (k > r->data) r->right = insert(r->right, k);
    else return r;

    r->height = 1 + max(h(r->left), h(r->right));
    int b = bal(r);

    if (b > 1 && k < r->left->data) return rRotate(r);
    if (b < -1 && k > r->right->data) return lRotate(r);
    if (b > 1 && k > r->left->data) { r->left = lRotate(r->left); return rRotate(r); }
    if (b < -1 && k < r->right->data) { r->right = rRotate(r->right); return lRotate(r); }

    return r;
}

void heapBottomUp(int H[], int n) {
    for (int k = n / 2; k >= 1; k--) {
        int v = H[k];
        int heap = 0;

        while (!heap && 2 * k <= n) {
            int j = 2 * k;

            if (j < n && H[j] < H[j + 1])
                j = j + 1;

            if (v >= H[j])
                heap = 1;
            else {
                H[k] = H[j];
                k = j;
            }
        }
        H[k] = v;
    }
}

int binarySearch(int a[], int n, int key) {
    int l = 0, r = n - 1;
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (a[mid] == key)
            return mid;
        else if (a[mid] < key)
            l = mid + 1;
        else
            r = mid - 1;
    }
    return -1; // not found
}
